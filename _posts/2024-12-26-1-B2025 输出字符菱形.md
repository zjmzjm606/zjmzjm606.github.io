#include<bits/stdc++.h>
#define For(k,i,j) for(int k=i;k<=j;k++)
using namespace std;
inline int read()
{
    char ch;
    int res=1;
    while((ch=getchar())<'0'||ch>'9') if(ch=='-') res=-1;
    res*=ch-48;
    while((ch=getchar())>='0'&&ch<='9')
        res=res*10+ch-48;
    return res;
}
int main()
{
    cout<<"  *  "<<endl;
    cout<<" ***  "<<endl;
    cout<<"*****"<<endl;
    cout<<" *** "<<endl;
    cout<<"  *  "<<endl;
    return 0;
}
