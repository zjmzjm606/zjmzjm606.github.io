#include<bits/stdc++.h>

using namespace std;
inline int read()
{
    char ch;
    int res=1;
    while((ch=getchar())<'0'||ch>'9') if(ch=='-') res=-1;
    res*=ch-48;
    while((ch=getchar())>='0'&&ch<='9')
        res=res*10+ch-48;
    return res;
}
int main()
{
    char n,ans;
    scanf("%c",&n);
    ans=char((int)n-32);
    printf("%c",ans);
    return 0;
}
