```cpp
#include<bits/stdc++.h>
#include<ctime>
using namespace std;
int main()
{
    srand(time(NULL));
    int a,b;
    long long ans;
    scanf("%d%d",&a,&b);
    while (ans!=a+b)
    {
        ans=rand()%2000000000+1;
    }
    printf("%d",ans);
    return 0;
}
```
