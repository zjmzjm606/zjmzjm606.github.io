```cpp
#include<iostream>
using namespace std;
int a[1000];
int main()
{
    
    int i,h,n=0;
    
    for(i=1;i<=10;i++)
      cin>>a[i];
    cin>>h;
    h+=30;
    for(i=1;i<=10;i++)
      if(h>=a[i])n++;
    cout<<n;  
    return 0;
}
```
