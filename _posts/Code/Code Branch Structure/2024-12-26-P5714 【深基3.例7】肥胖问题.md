```cpp
#include<iostream>
using namespace std;
int main()
{
    double n,m,ans;
    cin>>n>>m;
    ans=n/m/m;
    if(ans<18.5)
        cout<<"Underweight";
    else if(ans>=18.5&&ans<24)
        cout<<"Normal";
    else
    {
        printf("%.6g\n", ans);
        cout<<"Overweight";
    }    
    return 0;
}
```
