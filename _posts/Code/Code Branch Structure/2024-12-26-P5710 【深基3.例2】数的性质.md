```cpp
#include<iostream>
using namespace std;
int main()
{
    int n,sum=0;
    cin>>n;
    bool x=0,y=0;
    if(n%2==0)
        x=true,sum++;
    if(n>4&&n<=12)
        y=true,sum++;
    if(x==true&&y==true)
        cout<<1<<" ";
    else
        cout<<0<<" ";
    if(sum>=1)
        cout<<1<<" ";
    else
        cout<<0<<" ";
    if(sum==1)
        cout<<1<<" ";
    else
        cout<<0<<" ";
    if(sum==0)
        cout<<1<<endl;
    else
        cout<<0<<endl;
    return 0;
}
```
