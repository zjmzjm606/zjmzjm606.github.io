using namespace std;
inline int read()
{
    char ch;
    int res=1;
    while((ch=getchar())<'0'||ch>'9') if(ch=='-') res=-1;
    res*=ch-48;
    while((ch=getchar())>='0'&&ch<='9')
        res=res*10+ch-48;
    return res;
}
int main()
{
    int h,r;
    h=read();
    r=read();
    double v,ans;
    v=3.14*r*r*h;
    ans=20000/v;
    ans=int(ans)+1;
    cout<<ans;
    return 0;
}
