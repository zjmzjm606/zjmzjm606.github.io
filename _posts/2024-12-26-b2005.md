#include<bits/stdc++.h>

using namespace std;
inline int read()
{
    char ch;
    int res=1;
    while((ch=getchar())<'0'||ch>'9') if(ch=='-') res=-1;
    res*=ch-48;
    while((ch=getchar())>='0'&&ch<='9')
        res=res*10+ch-48;
    return res;
}
int main()
{
    char n;
    cin>>n;
    cout<<"  "<<n<<"  "<<endl;
    cout<<" "<<n<<n<<n<<" "<<endl;
    cout<<n<<n<<n<<n<<n<<endl;
    return 0;
}
